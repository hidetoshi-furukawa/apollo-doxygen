184 template <std::size_t N>
185 bool KalmanFilterConstState<N>::Predict(float delta_t) {
186  if (!inited_) {
187  return false;
188  }
189  predict_state_ = state_;
190  // state_transition_matrix is Identity
191  covariance_ += process_noise_;
192 
193  return true;
194 }

196 template <std::size_t N>
197 void KalmanFilterConstState<N>::Correct(const VectorNd &measurement) {
198  if (!inited_) {
199  Init(measurement);
200  } else {
201  auto measurements_cov = covariance_ + measure_noise_;
202  kalman_gain_ = covariance_ * measurements_cov.inverse();
203  state_ = state_ + kalman_gain_ * (measurement - state_);
204  covariance_ = (MatrixNd::Identity() - kalman_gain_) * covariance_;
205 
206  // compute likelihood
207  residual_ = measurement - predict_state_;
208  likelihood_ = std::exp(-0.5 * residual_.transpose() *
209  measurements_cov.inverse() * residual_) /
210  std::sqrt(2 * M_PI * measurements_cov.determinant());
211  }
212 }
