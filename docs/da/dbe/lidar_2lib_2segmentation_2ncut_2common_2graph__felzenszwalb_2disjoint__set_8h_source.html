75 int Universe::find(int x) {

84 void Universe::join(int x, int y) {
85  if (_elts[x].rank > _elts[y].rank) {
86  _elts[y].p = x;
87  _elts[x].size += _elts[y].size;
88  } else {
89  _elts[x].p = y;
90  _elts[y].size += _elts[x].size;
91  if (_elts[x].rank == _elts[y].rank) {
92  _elts[y].rank++;
93  }
