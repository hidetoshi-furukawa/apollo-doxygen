56 // Insertion sort
57 template <typename T, bool Compare(const T &, const T &)>
58 inline void IInsertionSort(T *a, int n) {
59  int i, j;
60  T tmp;
61  for (i = 1; i < n; i++) {
62  IMove(a[i], &tmp);
63  for (j = i; j > 0; j--) {
64  if (Compare(a[j - 1], tmp)) {
65  break;
66  }
67  IMove(a[j - 1], &a[j]);
68  }
69  IMove(tmp, &a[j]);
70  }
71 }
